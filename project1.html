#include "optimization.h"
#include <unordered_map>
#include <map>
#include <unordered_set>
#include <set>
using namespace std;
vector<int> mass;
unordered_map<int,set<int>> cnt;
map<int,unordered_set<int>> key;
void chng(int i,int j){
    int a=mass[i];
    key[cnt[a].size()].erase(a);
    key[cnt[j].size()].erase(j);
    cnt[a].erase(i);
    cnt[j].insert(i);
    key[cnt[j].size()].insert(j);
    key[cnt[a].size()].insert(a);
    mass[i]=j;
}

    int n;
    //cin>>n;
    n=readInt();
    mass.reserve(n);
    int num;
    for (int x=0;x<n;x++){
        //cin>>num;
        num=readInt();
        mass.push_back(num);
        key[cnt[num].size()].erase(num);
        cnt[num].insert(x);
        key[cnt[num].size()].insert(num);
    }
    int m;
    //cin>>m;
    m=readInt();
    char a;
    int i,j;
    for (int x=0;x<m;x++){
        //cin>>a;
        a=readChar();
        if (a=='?'){
            int y=0;
            key.erase(0);
            while (key.begin()->second.empty()){
                key.erase(key.begin());
            }
            if (key.begin()->second.size()>n/100){
                while (key.begin()->second.count(mass[y])==0){
                    y++;
                }
            }

                set<int> aba;
